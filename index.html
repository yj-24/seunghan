// 누적 참여 횟수 계산 함수 function getTotalDrawCount() { let currentTotal = items.reduce((sum, item) => sum + item.count, 0); return 1000 - currentTotal; // 시작이 1,000개이므로 } function startDraw() { const btn = document.getElementById('main-btn'); const status = document.getElementById('status-text'); const totalCount = items.reduce((sum, item) => sum + item.count, 0); if (totalCount <= 0) return; // 현재까지 몇 명이 참여했는지 계산 const drawCountSoFar = getTotalDrawCount(); btn.disabled = true; status.innerHTML = `
?
`; setTimeout(() => { let selected = null; // 300회 미만 참여 시 A상이 뽑히지 않도록 필터링 let availableItems = items.filter(item => { if (item.id === 'A' && drawCountSoFar < 500) { return false; // 500회 전까지는 리스트에서 제외 } return item.count > 0; }); // 필터링된 상품들 중에서의 새로운 총합 계산 const availableTotal = availableItems.reduce((sum, item) => sum + item.count, 0); let random = Math.floor(Math.random() * availableTotal); let currentSum = 0; for (let item of availableItems) { currentSum += item.count; if (random < currentSum) { selected = item; // 실제 원본 배열(items)에서 수량 차감 const originalItem = items.find(i => i.id === selected.id); originalItem.count--; break; } } saveData(); status.innerHTML = `
${selected.id}등급 당첨!
${selected.name}
`; render(); btn.disabled = false; }, 700); }
